**مەزمۇن قىسقىچە بايانى:** ياۋروپا ئىتتىپاقىنىڭ جەنۇبىي دېڭىز سىياسىتى 20-ئەسىرنىڭ 90-يىللىرىدىن جەنۇبىي دېڭىز كېسىمى دېلوسىدىن ئىلگىرىكى «پرىنسىپال بىتەرەپلىك» باسقۇچىنى ۋە جەنۇبىي دېڭىز كېسىمى دېلوسىدىن كېيىنكى «پرىنسىپال ئەمەلىيەتچىللىك» باسقۇچىنى باشتىن كەچۈردى. 2021-يىلى «ياۋروپا ئىتتىپاقىنىڭ ھىندى-تىنچ ئوكيان ھەمكارلىق ئىستراتېگىيەسى» ئېلان قىلىنغاندىن كېيىن، قايتا تەڭشەش باسقۇچىغا قەدەم قويدى. ياۋروپا ئىتتىپاقىنىڭ «ھىندى-تىنچ ئوكيان»غا بۇرۇلۇشى ئۇنىڭ جەنۇبىي دېڭىز سىياسىتىنىڭ لوگىكىسى، ئىدىيەسى ۋە يولىنىڭ تېخىمۇ تەرەققىي قىلغانلىقىدىن دېرەك بېرىدۇ. بۇ ئاساسلىقى جۇغراپىيەلىك بوشلۇقنىڭ بايان قىلىنىش ئۇسۇلىنىڭ ئۆزگىرىشى، «ئىستراتېگىيەلىك ئۆز-ئۆزىگە خوجا بولۇش» ئىدىيەسىنىڭ چوڭقۇر سىڭىپ كىرىشى ۋە ئارىلىشىش يوللىرىنىڭ كۆپ خىللىشىشىدا ئىپادىلىنىدۇ. بۇ ياۋروپا ئىتتىپاقىنىڭ گېئوپولىتىكىلىق رېئاللىققا بولغان تونۇشىنىڭ ئۆزگىرىشى، ئۆزىنىڭ ئىقتىسادىي ۋە بىخەتەرلىك قاتارلىق كۆپ تەرەپلىمىلىك مەنپەئەتىنى قوغداش ۋە «ئۆلچەملىك كۈچ» سۈپىتىدە يەرشارى دېڭىز-ئوكيان ئىدارە قىلىشقا قاتنىشىش قاتارلىق ئۈچ چوڭ ئامىلنىڭ بىرلىكتە تۈرتكە بولۇشىنىڭ نەتىجىسىدۇر. بۇنىڭدىن باشقا، ئۇ يەنە ئەزا دۆلەتلەرنىڭ تۈرتكىسى ۋە ئامېرىكا-ياۋروپا ئاتلانتىك ئوكيان ھالقىغان مۇناسىۋەت بېسىمىنىڭ تەسىرىگە ئۇچرىغان. ياۋروپا ئىتتىپاقىنىڭ جەنۇبىي دېڭىز سىياسىتىنىڭ ئاساسلىق مەزمۇنى ۋە قارار چىقىرىش ئۇسۇلىنى كۆزدە تۇتۇپ، خىتاي ئىقتىساد، سىياسەت، بىخەتەرلىك ۋە مۇھىت قاتارلىق ساھەلەردىكى ياۋروپا بىلەن بولغان ھەمكارلىقنى ماسلاشتۇرۇپ، جەنۇبىي دېڭىز ۋەزىيىتىنىڭ ياخشى تەرەپكە قاراپ تەرەققىي قىلىشىنى ئىلگىرى سۈرۈشى كېرەك.

**ئاچقۇچلۇق سۆزلەر:** ياۋروپا ئىتتىپاقى؛ جەنۇبىي دېڭىز؛ ھىندى-تىنچ ئوكيان ئىستراتېگىيەسى؛ ئىستراتېگىيەلىك ئۆز-ئۆزىگە خوجا بولۇش؛ گېئوپولىتىكا

# بىرىنچى، كىرىش سۆز

جەنۇبىي دېڭىز كېسىمى دېلوسى (2013-2016) نى چەك قىلغان ھالدا، ياۋروپا ئىتتىپاقىنىڭ جەنۇبىي دېڭىز مەسىلىسىگە ئارىلىشىش دەرىجىسى كۆرۈنەرلىك چوڭقۇرلاشتى. 2021-يىلى، ياۋروپا ئىتتىپاقى ئۆزىنىڭ ئاساسلىق ئەزا دۆلەتلىرى بولغان فرانسىيە، گېرمانىيە ۋە گوللاندىيەنىڭ تۈرتكىسىدە «ياۋروپا ئىتتىپاقىنىڭ ھىندى-تىنچ ئوكيان ھەمكارلىق ئىستراتېگىيەسى»نى ئېلان قىلدى. بۇنىڭدىن ئىلگىرى، ياۋروپا ئىتتىپاقىنىڭ جەنۇبىي دېڭىز سىياسىتى ئۇنىڭ ئاسىيا ئىستراتېگىيەسى ۋە شەرقىي ئاسىيا سىياسىتىگە تەۋە ئىدى، ئەمما ياۋروپا ئىتتىپاقىنىڭ «ھىندى-تىنچ ئوكيانغا بۇرۇلۇشى» ئۇنىڭ جەنۇبىي دېڭىز سىياسىتىنىڭ لوگىكىسى، ئىدىيەسى ۋە يولىدا تېخىمۇ چوڭقۇر ئىلگىرىلەش بولغانلىقىدىن دېرەك بېرىدۇ.

نۆۋەتتىكى تەتقىقاتلار ياۋروپا ئىتتىپاقىنىڭ جەنۇبىي دېڭىز مەسىلىسىگە ئارىلىشىشىغا دىققەت قىلغان بولۇپ، ئاساسلىقى ۋاقىت تەرتىپى بويىچە سىياسەتلەرنى رەتلىگەن. بەزى ئالىملار ياۋروپا ئىتتىپاقىنىڭ ئومۇمىي جەھەتتىن «پرىنسىپال بىتەرەپلىك» مەيدانىدا تۇرغانلىقىنى ۋە ئەمەلىيەتچىللىك ئالاھىدىلىكىنى نامايان قىلغانلىقىنى قارايدۇ. يەنە بەزى ئالىملار كېسىم دېلوسى مەزگىلىدە ياۋروپا ئىتتىپاقىنىڭ فىلىپپىننى قوللاش پوزىتسىيەسىدە بولغانلىقىنى، بۇنىڭ «بىتەرەپلىك»تىن ۋاقىتلىق چەتنەش ئىكەنلىكىنى بايقاشقان. ئەمما، «ھىندى-تىنچ ئوكيان ئىستراتېگىيەسى» ئارقا كۆرۈنۈشىدىكى ياۋروپا ئىتتىپاقىنىڭ جەنۇبىي دېڭىز سىياسىتى ئۈستىدە تېخى يېتەرلىك توختالمىغان.

شۇڭا، بۇ ماقالە 2021-يىلىدىكى ئىستراتېگىيە ئېلان قىلىنغاندىن كېيىنكى ئەھۋاللارنى ئاساس قىلىپ، خىتاينىڭ جەنۇبىي دېڭىز مەسىلىسىنى بىر تەرەپ قىلىشىغا پايدىلىق ماتېرىيال بىلەن تەمىنلەشنى مەقسەت قىلىدۇ.

# ئىككىنچى، ياۋروپا ئىتتىپاقى جەنۇبىي دېڭىز سىياسىتىنىڭ تەرەققىيات ئانالىزى

20-ئەسىرنىڭ 90-يىللىرىدىن ھازىرغىچە، ياۋروپا ئىتتىپاقىنىڭ جەنۇبىي دېڭىز سىياسىتى «پەرۋاسىزلىق»تىن «كۆڭۈل بۆلۈش»كە قاراپ بۇرۇلدى ۋە ئۈچ باسقۇچنى بېسىپ ئۆتتى:

1. **كېسىم دېلوسىدىن ئىلگىرىكى «پرىنسىپال بىتەرەپلىك» باسقۇچى:** بۇ مەزگىلدە ياۋروپا ئىتتىپاقى ئاساسلىقى ئىقتىسادىي مەنپەئەتكە ئەھمىيەت بەردى. 1994-يىلىدىكى «يېڭى ئاسىيا ئىستراتېگىيەسى»دە جەنۇبىي دېڭىز پەقەت بىر رايونلۇق توقۇنۇش نۇقتىسى سۈپىتىدە تىلغا ئېلىنغان. ياۋروپا ئىتتىپاقىنىڭ خىتاي بىلەن بولغان مۇناسىۋىتىدە ئىقتىسادىي ھەمكارلىق ئالدىنقى ئورۇندا تۇرۇپ، جەنۇبىي دېڭىز مەسىلىسىدە تىنچ سۆھبەتنى قوللايدىغانلىقىنى بىلدۈرۈپ كەلگەن.

2. **كېسىم دېلوسىدىن كېيىنكى «پرىنسىپال ئەمەلىيەتچىللىك» باسقۇچى:** 2016-يىلى فىلىپپىن قوزغىغان كېسىم دېلوسىدىن كېيىن، ياۋروپا ئىتتىپاقى خەلقئارا قانۇن ۋە دېڭىز-ئوكيان تەرتىپىنى تەكىتلەشكە باشلىدى. گەرچە «مەيدان تۇتمايمىز» دېسىمۇ، ئەمما كېسىم نەتىجىسىنى ئېتىراپ قىلىدىغانلىقىنى بىلدۈرۈش ئارقىلىق ئەمەلىيەتتە فىلىپپىن تەرەپتە تۇردى. شۇنىڭ بىلەن بىر ۋاقىتتا، خىتاي بىلەن بولغان ئىقتىسادىي مەنپەئەتنى كۆزلەپ، سۆھبەت ۋە ھەمكارلىقنىمۇ داۋاملاشتۇردى.

3. **«ھىندى-تىنچ ئوكيان ئىستراتېگىيەسى» ئاستىدىكى قايتا تەڭشەش باسقۇچى:** 2021-يىلىدىكى ئىستراتېگىيە بىلەن ياۋروپا ئىتتىپاقى جەنۇبىي دېڭىزنى ئۆزىنىڭ «ئىستراتېگىيەلىك ئۆز-ئۆزىگە خوجا بولۇش» نىشانىنىڭ بىر قىسمى قىلدى. جەنۇبىي دېڭىزنى يەرشارى سودىسى ۋە بىخەتەرلىكى ئۈچۈن ھالقىلىق دەپ بېكىتتى.

# ئۈچىنچى، ياۋروپا ئىتتىپاقى جەنۇبىي دېڭىز سىياسىتىنىڭ يېڭى مەزمۇنى ۋە سەۋەبلىرى

### يېڭى مەزمۇنلار:
1. **جۇغراپىيەلىك باياننىڭ ئۆزگىرىشى:** جەنۇبىي دېڭىزنى «ئاسىيا» چۈشەنچىسىدىن چىقىرىپ، تېخىمۇ كەڭ بولغان «ھىندى-تىنچ ئوكيان» رامكىسىغا كىرگۈزدى. بۇ ئۇنىڭ دېڭىز-ئوكيان خۇسۇسىيىتىنى تېخىمۇ گەۋدىلەندۈردى.
2. **«ئىستراتېگىيەلىك ئۆز-ئۆزىگە خوجا بولۇش»:** ياۋروپا ئىتتىپاقى ئەمدى بىخەتەرلىك جەھەتتە پۈتۈنلەي ئامېرىكىغا تايانماي، ئۆزىنىڭ مۇستەقىل ھەرىكەت قىلىش ئىقتىدارىنى يېتىلدۈرمەكچى.
3. **ئارىلىشىش يوللىرىنىڭ كۆپ خىللىشىشى:** سىياسىي باياناتلاردىن باشقا، كىلىمات ئۆزگىرىشى، بېلىقچىلىقنى باشقۇرۇش ۋە ئوخشاش كۆز قاراشتىكى دۆلەتلەر (ياپونىيە، ئاۋسترالىيە قاتارلىق) بىلەن ھەمكارلىشىش ئارقىلىق ئارىلىشىشنى كۈچەيتمەكتە.

### سەۋەبلەر:
1. **گېئوپولىتىكىلىق تونۇشنىڭ ئۆزگىرىشى:** دۇنيا مەركىزىنىڭ ھىندى-تىنچ ئوكيانغا يۆتكىلىشى بىلەن ياۋروپا ئۆزىنى بىر گېئوپولىتىكىلىق كۈچ سۈپىتىدە نامايان قىلىشقا مەجبۇر بولدى.
2. **ئىقتىسادىي ۋە بىخەتەرلىك مەنپەئەتى:** ياۋروپانىڭ تاشقى سودىسىنىڭ 40% ى جەنۇبىي دېڭىزدىن ئۆتىدۇ. بۇ رايوننىڭ مۇقىملىقى ياۋروپانىڭ تۇرمۇش لىنىيەسىگە چېتىلىدۇ.
3. **ئۆلچەملىك كۈچ سۈپىتىدىكى رولى:** ياۋروپا ئىتتىپاقى خەلقئارا دېڭىز-ئوكيان قانۇنىنى قوغدىغۇچى سۈپىتىدە، ئۆزىنىڭ قىممەت قارىشىنى يەرشارى مىقياسىدا كېڭەيمەكچى.
4. **ئەزا دۆلەتلەر ۋە ئامېرىكىنىڭ بېسىمى:** فرانسىيە، گېرمانىيە قاتارلىق دۆلەتلەرنىڭ ئۆز ئالدىغا تۈزگەن ئىستراتېگىيەلىرى ۋە ئامېرىكىنىڭ ئاتلانتىك ئوكيان ھالقىغان ئىتتىپاقداشلىق بېسىمى ياۋروپا ئىتتىپاقىنى تېخىمۇ ئاكتىپ پوزىتسىيە تۇتۇشقا ئىتتەردى.

# تۆتىنچى، خۇلاسە

جەنۇبىي دېڭىزنىڭ ئىستراتېگىيەلىك ئەھمىيىتىنىڭ ئېشىشىغا ئەگىشىپ، ياۋروپا ئىتتىپاقىنىڭ پوزىتسىيەسىمۇ زور دەرىجىدە ئۆزگەردى. گەرچە ئۇكرائىنا كىرىزىسى تۈپەيلىدىن دىققەت مەركىزى ۋاقىتلىق قوشنا رايونلارغا يۆتكەلگەن بولسىمۇ، ئەمما جەنۇبىي دېڭىز ياۋروپا ئىتتىپاقىنىڭ ھىندى-تىنچ ئوكياندىكى تەسىر كۈچىنى ئاشۇرۇشنىڭ مۇھىم تايانچ نۇقتىسى بولۇپ قالىدۇ.

خىتاي ياۋروپا ئىتتىپاقىنىڭ سىياسىي قارار چىقىرىش مېخانىزمىنى (مەسىلەن، ئەزا دۆلەتلەر ئارا پىكىر بىرلىكى ۋە ياۋروپا ئىتتىپاقىنىڭ ئومۇمىي سىياسىتى) ئېنىق چۈشىنىشى ھەمدە ئىقتىساد، سىياسەت، بىخەتەرلىك ۋە مۇھىت قاتارلىق ساھەلەردە ياۋروپا بىلەن ماسلىشىپ ھەرىكەت قىلىشى كېرەك. بۇ ئارقىلىق جەنۇبىي دېڭىز ۋەزىيىتىنىڭ يەنىمۇ جىددىيلىشىپ كېتىشىنىڭ ئالدىنى ئېلىشقا بولىدۇ.
